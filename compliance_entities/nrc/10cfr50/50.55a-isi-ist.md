## Overview

§50.55a requires nuclear power plants to comply with the ASME Boiler and Pressure Vessel (B&PV) Code and the ASME Code for Operation and Maintenance of Nuclear Power Plants (OM Code) for:

- **ISI (Inservice Inspection):** Nondestructive examination of pressure-retaining components — reactor vessel, piping, pumps, valves, and supports — per ASME B&PV Code Section XI
- **IST (Inservice Testing):** Functional testing of pumps, valves, and dynamic restraints to verify operability — per ASME OM Code

The 10-year ISI inspection interval and the quarterly/2-year/5-year IST intervals are the primary **DETERMINISTIC** obligations in this section. Program scope and NDE technique qualification are **PARAMETERIZED** or **CONTESTED**.

### Tests — interval and period compliance

```python
import pytest
from datetime import date
from dateutil.relativedelta import relativedelta

INTERVAL_MONTHS = 120           # 10-year ISI interval
MAX_EXTENSION_MONTHS = 12       # Without NRC approval (ASME XI IWA-2430(d))

# Approximate inspection period distribution within a 10-year interval
PERIOD_DISTRIBUTION = {
    1: 0.40,   # 40% of total required inspections
    2: 0.20,   # 20% of total required inspections
    3: 0.40,   # 40% of total required inspections
}


class TestISIIntervalCompliance:
    """Pattern 1: DETERMINISTIC — interval timing is unambiguous."""

    @pytest.mark.assumption(
        id="ASSUME-NRC50-ISI-002",
        description=(
            "10-year interval start: interval begins at commercial operation date (date of "
            "first fuel load for operating licenses, or ITAAC completion date for Part 52 "
            "combined licenses). Subsequent intervals begin at the end of the preceding "
            "10-year interval. Extensions up to 12 months are allowed without NRC approval per "
            "ASME XI IWA-2430(d); extensions beyond 12 months require §50.55a(z) alternative request."
        ),
        approved_by="ISI Program Engineer",
        review_date="2026-05-21",
    )
    def test_current_isi_interval_not_expired(self, isi_interval: dict):
        interval_start = isi_interval.get("interval_start_date")
        assert interval_start is not None, (
            "ISI interval has no documented start date — cannot verify 10-year interval compliance"
        )
        assert isinstance(interval_start, date), "interval_start_date must be a date object"

        interval_end = interval_start + relativedelta(months=INTERVAL_MONTHS)
        max_extended_end = interval_end + relativedelta(months=MAX_EXTENSION_MONTHS)

        today = date.today()
        assert today <= max_extended_end, (
            f"10-year ISI interval expired without completion: interval began {interval_start}, "
            f"nominal end {interval_end}, maximum extended end {max_extended_end} — "
            f"today is {today}. Extensions >12 months require NRC §50.55a(z) alternative request."
        )

    def test_isi_interval_end_date_documented(self, isi_interval: dict):
        assert isi_interval.get("interval_end_date") is not None, (
            "ISI program does not document interval end date — required for scheduling outage "
            "work and verifying inspection period compliance"
        )
        assert isi_interval.get("current_interval_number") is not None, (
            "ISI program does not identify which 10-year interval the plant is in (1st, 2nd, 3rd, etc.)"
        )

    def test_isi_interval_extension_has_nrc_approval_if_needed(self, isi_interval: dict):
        interval_start = isi_interval.get("interval_start_date")
        nominal_end = isi_interval.get("nominal_interval_end_date")
        actual_end = isi_interval.get("actual_interval_end_date") or date.today()

        if interval_start and nominal_end and isinstance(nominal_end, date) and isinstance(actual_end, date):
            extension_days = (actual_end - nominal_end).days
            extension_months = extension_days / 30.44

            if extension_months > MAX_EXTENSION_MONTHS:
                nrc_alternative = isi_interval.get("nrc_alternative_request_number")
                assert nrc_alternative is not None, (
                    f"ISI interval extended {extension_months:.1f} months beyond nominal end "
                    f"(>{MAX_EXTENSION_MONTHS} months) without an NRC §50.55a(z) alternative "
                    "request on record — this requires NRC approval"
                )


class TestISIInspectionPeriods:
    """Pattern 2: PARAMETERIZED — percentage completion requires program data."""

    @pytest.mark.assumption(

            "target within each period is: Period 1: 40%, Period 2: 20%, Period 3: 40% of all "

            assert period.get("period_number") is not None
            assert period.get("start_date") is not None
```

```python
class TestISTPumpAcceptanceCriteria:
    """Pattern 1: DETERMINISTIC — ASME OM Table ISTB-5121-1 thresholds are fixed."""

    def test_pump_differential_pressure_within_limits(self, pump_test_result: dict):
        reference_dp = pump_test_result.get("reference_differential_pressure_psid")
        measured_dp = pump_test_result.get("measured_differential_pressure_psid")

        if reference_dp is None or measured_dp is None:
            pytest.skip("Differential pressure reference or measurement not recorded — "
                        "data quality issue; requires IST program engineer review")

        decrease_fraction = (reference_dp - measured_dp) / reference_dp if reference_dp > 0 else 0

        assert decrease_fraction < PUMP_DIFFERENTIAL_PRESSURE_REQUIRED_ACTION_DECREASE_THRESHOLD, (
            f"Pump '{pump_test_result.get('pump_id')}': differential pressure decrease of "
            f"{decrease_fraction:.1%} exceeds Required Action level "
            f"({PUMP_DIFFERENTIAL_PRESSURE_REQUIRED_ACTION_DECREASE_THRESHOLD:.0%}) — "
            "ASME OM Table ISTB-5121-1: pump shall be declared inoperable, investigation required, "
            "and corrective action taken before return to service"
        )

        if decrease_fraction >= PUMP_DIFFERENTIAL_PRESSURE_ALERT_DECREASE_THRESHOLD:
            # Alert level — not a failure but requires increased surveillance
            # Record alert in test data for trending; do not fail the assertion
            pump_test_result["alert_flag_differential_pressure"] = True

    def test_pump_vibration_within_limits(self, pump_test_result: dict):
        reference_vibration = pump_test_result.get("reference_vibration_in_per_sec")
        measured_vibration = pump_test_result.get("measured_vibration_in_per_sec")

        if reference_vibration is None or measured_vibration is None:
            pytest.skip("Vibration reference or measurement not recorded")

        vibration_multiplier = (
            measured_vibration / reference_vibration if reference_vibration > 0 else 0
        )

        assert vibration_multiplier < PUMP_VIBRATION_REQUIRED_ACTION_MULTIPLIER, (
            f"Pump '{pump_test_result.get('pump_id')}': vibration is {vibration_multiplier:.1f}× "
            f"reference value, exceeding Required Action level ({PUMP_VIBRATION_REQUIRED_ACTION_MULTIPLIER}×) — "
            "ASME OM Table ISTB-5121-1: pump shall be declared inoperable; corrective action required"
        )

        if vibration_multiplier >= PUMP_VIBRATION_ALERT_MULTIPLIER:
            pump_test_result["alert_flag_vibration"] = True
```

```python
import pytest
from datetime import date
from dateutil.relativedelta import relativedelta

CATEGORY_A_B_QUARTERLY_INTERVAL_MONTHS = 3
CATEGORY_A_B_COLD_SHUTDOWN_MAX_MONTHS = 60   # 5 years maximum
SAFETY_RELIEF_VALVE_TEST_INTERVAL_MONTHS = 60  # 5 years

# Stroke time acceptance levels per ASME OM ISTC-5221-1
VALVE_STROKE_TIME_ALERT_FRACTION = 0.50        # ±50% from reference = Alert
VALVE_STROKE_TIME_REQUIRED_ACTION_FRACTION = 1.00  # ±100% (doubled or halved) = Required Action


class TestISTValveIntervals:
    """Pattern 1: DETERMINISTIC — valve test intervals are fixed by ASME OM ISTC."""

    @pytest.mark.assumption(
        id="ASSUME-NRC50-ISI-006",
        description=(
            "IST valve categories: Category A valves have a safety function in only the open "
            "or closed position; Category B have safety function in both positions; Category C "
            "are check valves with a safety function. Valve category is assigned in the plant IST "
            "program and approved by the AAII. Power-operated valves (Category A/B) that cannot "
            "be exercised during normal operation without disabling a safety function are deferred "
            "to cold shutdown or refueling outage, maximum 5 years or 2 outages."
        ),
        approved_by="IST Program Engineer",
        review_date="2026-05-21",
    )
    def test_category_a_b_valve_exercise_not_overdue(self, valve_record: dict):
        category = valve_record.get("category")
        if category not in ("A", "B"):
            pytest.skip("This test applies to Category A and B valves only")

        can_test_normal_ops = valve_record.get("can_exercise_during_normal_ops", True)

        if can_test_normal_ops:
            last_test_date = valve_record.get("last_quarterly_exercise_date")
            interval_months = CATEGORY_A_B_QUARTERLY_INTERVAL_MONTHS
            interval_description = "quarterly (3-month)"
        else:
            last_test_date = valve_record.get("last_cold_shutdown_exercise_date")
            interval_months = CATEGORY_A_B_COLD_SHUTDOWN_MAX_MONTHS
            interval_description = "cold-shutdown maximum (5-year)"

        assert last_test_date is not None, (
            f"Valve '{valve_record.get('valve_id')}': no {interval_description} IST exercise "
            "on record — Category A/B valve exercise interval per ASME OM ISTC is a "
            "10 CFR §50.55a(f)(4)(i) requirement"
        )

        due_date = last_test_date + relativedelta(months=interval_months)
        today = date.today()
        assert today <= due_date, (
            f"Valve '{valve_record.get('valve_id')}' (Category {category}): {interval_description} "
            f"exercise overdue — last exercise: {last_test_date}, due: {due_date}, today: {today}"
        )

    def test_safety_relief_valve_setpoint_test_not_overdue(self, valve_record: dict):
        if not valve_record.get("is_safety_relief_valve", False):
            pytest.skip("This test applies to safety/relief valves (PRDs) only")

        last_test_date = valve_record.get("last_setpoint_test_date")
        assert last_test_date is not None, (
            f"Safety/relief valve '{valve_record.get('valve_id')}': no setpoint test on record — "
            "ASME OM ISTC-3600 requires PRD setpoint testing on a 5-year interval"
        )

        due_date = last_test_date + relativedelta(months=SAFETY_RELIEF_VALVE_TEST_INTERVAL_MONTHS)
        today = date.today()
        assert today <= due_date, (
            f"Safety/relief valve '{valve_record.get('valve_id')}': 5-year setpoint test overdue — "
            f"last test: {last_test_date}, due: {due_date}, today: {today}"
        )

    def test_safety_relief_valve_as_left_setpoint_within_tolerance(self, valve_record: dict):
        if not valve_record.get("is_safety_relief_valve", False):
            pytest.skip("This test applies to safety/relief valves only")

        nominal_setpoint_psig = valve_record.get("nominal_setpoint_psig")
        as_left_setpoint_psig = valve_record.get("as_left_setpoint_psig")

        if nominal_setpoint_psig is None or as_left_setpoint_psig is None:
            pytest.skip("Safety/relief valve setpoint values not recorded")

        tolerance_fraction = 0.01  # ±1% of nominal
        deviation = abs(as_left_setpoint_psig - nominal_setpoint_psig) / nominal_setpoint_psig

        assert deviation <= tolerance_fraction, (
            f"Safety/relief valve '{valve_record.get('valve_id')}': as-left setpoint "
            f"{as_left_setpoint_psig} psig deviates {deviation:.2%} from nominal "
            f"{nominal_setpoint_psig} psig — ASME OM ISTC-3600 requires as-left within ±1% "
            "of nominal setpoint; valve must be adjusted before return to service"
        )


class TestISTValveStrokeTimes:
    """Pattern 1: DETERMINISTIC — stroke time Required Action levels are fixed."""

    def test_valve_stroke_time_within_required_action_level(self, valve_test_result: dict):
        category = valve_test_result.get("category")
        if category not in ("A", "B"):
            pytest.skip("Stroke time testing applies to Category A and B power-operated valves")

        reference_stroke_time_sec = valve_test_result.get("reference_stroke_time_sec")
        measured_stroke_time_sec = valve_test_result.get("measured_stroke_time_sec")

        if reference_stroke_time_sec is None or measured_stroke_time_sec is None:
            pytest.skip("Stroke time reference or measurement not recorded")

        # Check for both lengthening and shortening per ASME OM ISTC-5221-1
        fractional_deviation = (
            abs(measured_stroke_time_sec - reference_stroke_time_sec) / reference_stroke_time_sec
            if reference_stroke_time_sec > 0 else 0
        )

        assert fractional_deviation < VALVE_STROKE_TIME_REQUIRED_ACTION_FRACTION, (
            f"Valve '{valve_test_result.get('valve_id')}' (Category {category}): stroke time "
            f"{measured_stroke_time_sec:.1f}s deviates {fractional_deviation:.0%} from reference "
            f"{reference_stroke_time_sec:.1f}s — exceeds Required Action level "
            f"({VALVE_STROKE_TIME_REQUIRED_ACTION_FRACTION:.0%} from reference) per "
            "ASME OM ISTC-5221-1; valve shall be declared inoperable until corrective action taken"
        )

        if fractional_deviation >= VALVE_STROKE_TIME_ALERT_FRACTION:
            valve_test_result["alert_flag_stroke_time"] = True
```
