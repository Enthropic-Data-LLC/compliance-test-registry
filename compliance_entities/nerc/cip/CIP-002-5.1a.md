## Scope summary

CIP-002 establishes the foundational categorization that scopes every other CIP standard. Each Responsible Entity must identify its BES Cyber Systems and assign each an impact rating (High, Medium, or Low) per the bright-line criteria in Attachment 1.

The standard is unusually deterministic by NERC standards — Attachment 1 provides numerical thresholds (MW, kV, etc.) that translate directly to boolean tests. The principal ambiguity historically has been around what constitutes a "BES Cyber System" boundary, which CIP-002-8 (filed but not yet enforceable) tightens further.

## R1 — Identification and Categorization

### Source excerpt (R1)

> Each Responsible Entity shall implement a process that considers each of the following assets for purposes of Parts 1.1 through 1.3: Control Centers and backup Control Centers; Transmission stations and substations; Generation resources; Systems and facilities critical to system restoration; Special Protection Systems; and Distribution Provider Protection Systems. The Responsible Entity shall identify each of the high impact BES Cyber Systems (Part 1.1), medium impact BES Cyber Systems (Part 1.2), and if any, low impact BES Cyber Systems (Part 1.3), according to Attachment 1.

## Generated tests (Pattern 1 — Direct Assertion)

### `tests/CIP-002/test_R1.py`

```python
"""
CIP-002-5.1a R1 — BES Cyber System Categorization
Confidence: HIGH | Human Review: NOT REQUIRED
"""
import pytest
from compliance.criteria import attachment_1_matches

HIGH_IMPACT_ROLES = {
    "Reliability Coordinator",
    "Balancing Authority",
}


def test_CIP002_R1_1_high_impact_control_centers(live_assets):
    """R1.1: Control Centers performing RC/BA functions => High Impact."""
    for asset in live_assets:
        if asset["asset_type"] == "Control Center" and \
           set(asset.get("functional_roles", [])).intersection(HIGH_IMPACT_ROLES):
            assert asset["impact_rating"] == "High", (
                f"CRITICAL VIOLATION: {asset['asset_id']} performs "
                f"{HIGH_IMPACT_ROLES.intersection(asset['functional_roles'])} "
                f"but is rated {asset['impact_rating']}, not High."
            )


def test_CIP002_R1_2_medium_impact_thresholds(live_assets):
    """R1.2: Medium impact via Attachment 1 Section 2 bright-lines."""
    for asset in live_assets:
        match = attachment_1_matches(asset, section=2)
        if match.matched:
            assert asset["impact_rating"] == "Medium", (
                f"VIOLATION: {asset['asset_id']} matched Attachment 1 "
                f"criterion {match.criterion_id} ({match.rationale}) "
                f"but is rated {asset['impact_rating']}, not Medium."
            )


def test_CIP002_R1_3_low_impact_completeness(live_assets, in_scope_sites):
    """R1.3: Every in-scope asset must have an impact rating assigned."""
    for site in in_scope_sites:
        rated_assets = [a for a in live_assets if a["site_id"] == site["site_id"]]
        unrated = [a for a in rated_assets if not a.get("impact_rating")]
        assert not unrated, (
            f"VIOLATION: Site {site['site_id']} has {len(unrated)} "
            f"unrated BES Cyber Systems: {[a['asset_id'] for a in unrated]}"
        )


def test_CIP002_R1_rating_consistency(live_assets):
    """No asset may be rated Low if it matches a Medium or High criterion."""
    for asset in live_assets:
        if asset["impact_rating"] == "Low":
            for section in (1, 2):
                match = attachment_1_matches(asset, section=section)
                assert not match.matched, (
                    f"MISCATEGORIZATION: {asset['asset_id']} is rated Low "
                    f"but matches Attachment 1 Section {section} "
                    f"criterion {match.criterion_id}."
                )
```

### `tests/CIP-002/test_R2.py`

```python
"""
CIP-002-5.1a R2 — Annual Review and Approval
Confidence: HIGH | Human Review: NOT REQUIRED
"""
from datetime import datetime, timedelta

# 15 calendar months expressed as a conservative day count (15 * 30.44 ≈ 457)
REVIEW_WINDOW_DAYS = 457


def test_CIP002_R2_1_review_cadence(categorization_review):
    """R2.1: R1 identifications reviewed within 15 calendar months."""
    age = datetime.utcnow() - categorization_review.last_reviewed_at
    assert age <= timedelta(days=REVIEW_WINDOW_DAYS), (
        f"VIOLATION: Categorization review is {age.days} days old. "
        f"Threshold: {REVIEW_WINDOW_DAYS} days. "
        f"Last reviewed: {categorization_review.last_reviewed_at.isoformat()}"
    )


def test_CIP002_R2_2_senior_manager_approval(categorization_review):
    """R2.2: Approval by CIP Senior Manager or named delegate."""
    assert categorization_review.approved_by_role in (
        "CIP Senior Manager",
        "CIP Senior Manager Delegate",
    ), (
        f"VIOLATION: Categorization approved by "
        f"'{categorization_review.approved_by_role}', which is not an "
        f"authorized role."
    )
    assert categorization_review.signature_hash, (
        "INTEGRITY FAILURE: Categorization approval lacks signature hash."
    )

    age = datetime.utcnow() - categorization_review.approved_at
    assert age <= timedelta(days=REVIEW_WINDOW_DAYS), (
        f"VIOLATION: Approval is {age.days} days old (threshold "
        f"{REVIEW_WINDOW_DAYS}). Re-approval required."
    )
```
