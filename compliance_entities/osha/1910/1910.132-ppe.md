## Scope summary

1910.132 establishes general requirements for PPE used in work environments where hazards cannot be eliminated through engineering or administrative controls. It covers protection for: eyes and face (1910.133), head (1910.135), foot (1910.136), hand (1910.138), and hearing (1910.95). The core requirements are: (d) documented hazard assessment with certification, (f) employee training, and (h) employer payment for required PPE. Specific PPE types are governed by separate subpart I sections.

---

## (d) — Hazard Assessment and Equipment Selection

### Source excerpt

> *The employer shall assess the workplace to determine if hazards are present, or are likely to be present, which necessitate the use of personal protective equipment. If such hazards are present, or likely to be present, the employer shall: select, and have each affected employee use, the types of PPE that will protect the affected employee from the hazards identified; communicate selection decisions to each affected employee; and select PPE that properly fits each affected employee.*

> *The employer shall verify that the required workplace hazard assessment has been performed through a written certification that identifies the workplace evaluated, the person certifying that the evaluation has been performed, the date(s) of the hazard assessment, and which identifies the document as a certification of hazard assessment.*

## (f) — Training

### Source excerpt

> *The employer shall provide training to each employee who is required by this section to use PPE. Each such employee shall be trained to know at least the following: when PPE is necessary; what PPE is necessary; how to properly don, doff, adjust, and wear PPE; the limitations of the PPE; the proper care, maintenance, useful life and disposal of the PPE.*

> *Retrain the employee when there is reason to believe the employee does not have the understanding and skill required, when changes in the workplace render previous training obsolete, or when changes in the types of PPE render previous training obsolete.*

## Generated tests

### `tests/1910.132/test_d_hazard_assessment.py`

```python
"""
1910.132(d) — Hazard Assessment and Certification
Confidence: MEDIUM | Human Review: ASSUMPTION REQUIRED (ASSUME-132-001)
"""
import pytest
from datetime import datetime, timedelta

ASSESSMENT_REVIEW_INTERVAL_DAYS = 365


@pytest.mark.assumption(
    id="ASSUME-132-001",
    description=(
        "Hazard assessment is adequate when it documents tasks, hazard categories, "
        "PPE selection basis, and is reviewed when processes change"
    ),
    approved_by="Compliance Officer",
    review_date="2026-05-20",
)
def test_hazard_assessment_exists_for_all_work_areas(
    work_areas, hazard_assessments
):
    assessed_area_ids = {h["workplace_id"] for h in hazard_assessments}
    unassessed = [
        a for a in work_areas
        if a["area_id"] not in assessed_area_ids and a.get("has_ppe_hazards")
    ]
    assert not unassessed, (
        f"VIOLATION: {len(unassessed)} work area(s) with PPE hazards have no "
        f"hazard assessment: {[a['area_id'] for a in unassessed]}"
    )


@pytest.mark.assumption(id="ASSUME-132-001")
def test_hazard_assessment_certification_fields_complete(hazard_assessments):
    required_fields = {"workplace_id", "certifier_name", "assessment_date", "certified_document_flag"}
    violations = []
    for ha in hazard_assessments:
        missing = {f for f in required_fields if not ha.get(f)}
        if missing:
            violations.append(
                f"Assessment {ha.get('assessment_id', '?')}: missing {missing}"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} hazard assessment(s) missing required "
        f"certification fields:\n" + "\n".join(violations)
    )


@pytest.mark.assumption(id="ASSUME-132-001")
def test_hazard_assessment_current(hazard_assessments):
    violations = []
    for ha in hazard_assessments:
        age = datetime.utcnow().date() - ha["assessment_date"]
        if age.days > ASSESSMENT_REVIEW_INTERVAL_DAYS:
            violations.append(
                f"Area {ha['workplace_id']}: assessment is {age.days} days old "
                f"(threshold {ASSESSMENT_REVIEW_INTERVAL_DAYS})"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} hazard assessment(s) require re-review:\n"
        + "\n".join(violations)
    )
```

### `tests/1910.132/test_f_training.py`

```python
"""
1910.132(f) — PPE Training
Confidence: HIGH | Human Review: NOT REQUIRED
"""

REQUIRED_TRAINING_TOPICS = {
    "when_ppe_necessary",
    "what_ppe_required",
    "donning_doffing",
    "limitations",
    "care_maintenance_disposal",
}


def test_ppe_training_completed_before_first_use(
    ppe_assignments, training_records
):
    violations = []
    for assignment in ppe_assignments:
        records = [
            r for r in training_records
            if r["employee_id"] == assignment["employee_id"]
            and r["standard_id"] == "1910.132"
        ]
        if not records:
            violations.append(
                f"{assignment['employee_id']}: no PPE training on record"
            )
            continue
        earliest = min(r["training_date"] for r in records)
        if earliest > assignment.get("first_ppe_use_date", earliest):
            violations.append(
                f"{assignment['employee_id']}: PPE training ({earliest}) "
                f"after first use ({assignment['first_ppe_use_date']})"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} employee(s) lack timely PPE training:\n"
        + "\n".join(violations)
    )


def test_ppe_training_covers_required_topics(training_records):
    violations = []
    for r in training_records:
        if r["standard_id"] != "1910.132":
            continue
        covered = set(r.get("topics_covered", []))
        missing = REQUIRED_TRAINING_TOPICS - covered
        if missing:
            violations.append(
                f"{r['employee_id']} on {r['training_date']}: missing {missing}"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} PPE training record(s) missing topics:\n"
        + "\n".join(violations)
    )
```

"""
1910.132(h) — Employer Payment for PPE
Confidence: HIGH | Human Review: NOT REQUIRED
"""

EMPLOYER_PAY_EXCEPTIONS = {
    "non_specialty_safety_toe_footwear",
    "non_specialty_prescription_eyewear",
    "logging_boots",
    "employee_choice_own_equipment",
}


def test_required_ppe_provided_by_employer(ppe_procurement_records):
    violations = [
        r for r in ppe_procurement_records
        if r.get("ppe_required")
        and r.get("paid_by") != "employer"
        and r.get("exception_category") not in EMPLOYER_PAY_EXCEPTIONS
    ]
    assert not violations, (
        f"VIOLATION: {len(violations)} required PPE item(s) not provided by "
        f"employer without a documented exception: "
        f"{[r.get('ppe_id') for r in violations]}"
    )

## Notes for the registry

- **Subpart I cross-references:** 1910.132 is the umbrella; specific equipment requirements live in: 1910.133 (eye/face), 1910.135 (head), 1910.136 (foot), 1910.137 (electrical protective equipment), 1910.138 (hand), 1910.95 (hearing). Tests for specific equipment types should live in their respective files and reference the hazard assessment data from this standard.
- **PPE fit (d)(1)(iii):** PPE must properly fit. This is not just selection — it requires a documented fit check for each individual employee assigned the PPE. Consider adding a `fit_verified` field to the `ppe_assignment` schema.
- **Defective PPE:** (d)(3) requires defective or damaged PPE to be removed from service. Add a test against `ppe_inventory.status != "defective_in_service"`.
