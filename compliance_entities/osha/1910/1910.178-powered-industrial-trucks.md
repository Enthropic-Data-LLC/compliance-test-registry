## Scope summary

1910.178 governs the design, maintenance, and operation of forklifts and other powered industrial trucks (PITs) in general industry workplaces. The 1999 revision significantly strengthened operator training requirements. The most-cited provisions are (l) (operator training and evaluation), (m) (truck operations — prohibited actions), and (q) (maintenance). The standard incorporates ASME B56.1 (Safety Standard for Low Lift and High Lift Trucks) by reference.

### `tests/1910.178/test_l_training.py`

```python
"""
1910.178(l) — PIT Operator Training
Confidence: HIGH | Human Review: NOT REQUIRED
"""
from datetime import datetime, timedelta

TRIENNIAL_EVALUATION_DAYS = 1095

REQUIRED_TRAINING_TOPICS = {
    "load_capacity",
    "stability_triangle",
    "pre_operation_inspection",
    "refueling_recharging",
    "controls_instrumentation",
    "surface_conditions",
    "ramps_and_inclines",
    "hazardous_locations",
    "load_handling",
    "pedestrian_traffic",
    "prohibited_actions",
}


def test_all_pit_operators_trained_before_first_operation(
    pit_operators, pit_training_records
):
    violations = []
    for op in pit_operators:
        records = [
            r for r in pit_training_records
            if r["employee_id"] == op["employee_id"]
        ]
        if not records:
            violations.append(
                f"{op['employee_id']}: no PIT training on record"
            )
            continue
        earliest = min(r["training_date"] for r in records)
        if earliest > op.get("first_operation_date", earliest):
            violations.append(
                f"{op['employee_id']}: trained {earliest} AFTER first operation "
                f"{op['first_operation_date']}"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} PIT operator(s) without pre-operation training:\n"
        + "\n".join(violations)
    )


def test_pit_training_includes_practical_evaluation(pit_training_records):
    violations = [
        r for r in pit_training_records
        if not r.get("practical_evaluation_passed")
    ]
    assert not violations, (
        f"VIOLATION: {len(violations)} PIT training record(s) missing practical "
        f"evaluation: {[r['employee_id'] for r in violations]}"
    )


def test_pit_triennial_evaluation_current(pit_operators, pit_evaluation_records):
    violations = []
    for op in pit_operators:
        evals = [
            r for r in pit_evaluation_records
            if r["employee_id"] == op["employee_id"]
        ]
        if not evals:
            violations.append(f"{op['employee_id']}: no PIT evaluation on record")
            continue
        most_recent = max(evals, key=lambda r: r["evaluation_date"])
        age = datetime.utcnow().date() - most_recent["evaluation_date"]
        if age.days > TRIENNIAL_EVALUATION_DAYS:
            violations.append(
                f"{op['employee_id']}: last PIT evaluation {age.days} days ago "
                f"(threshold {TRIENNIAL_EVALUATION_DAYS})"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} PIT operator(s) overdue for triennial evaluation:\n"
        + "\n".join(violations)
    )
```

### `tests/1910.178/test_q_maintenance.py`

"""
1910.178(q) — Maintenance and Pre-Shift Inspection
Confidence: HIGH | Human Review: NOT REQUIRED
"""
from datetime import datetime, timedelta


def test_pre_shift_inspection_completed_for_all_active_trucks(
    pit_inventory, pre_shift_inspections
):
    today = datetime.utcnow().date()
    active_trucks = [t for t in pit_inventory if t.get("status") == "active"]
    violations = []
    for truck in active_trucks:
        today_inspections = [
            i for i in pre_shift_inspections
            if i["truck_id"] == truck["truck_id"]
            and i["inspection_date"] == today
        ]
        if not today_inspections:
            violations.append(
                f"Truck {truck['truck_id']}: no pre-shift inspection today"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} active PIT(s) without today's pre-shift "
        f"inspection:\n" + "\n".join(violations)
    )


def test_defective_trucks_removed_from_service(pre_shift_inspections, pit_inventory):
    violations = []
    for inspection in pre_shift_inspections:
        if inspection.get("defects_found") and not inspection.get("truck_removed_from_service"):
            truck = next(
                (t for t in pit_inventory
                 if t["truck_id"] == inspection["truck_id"]),
                None,
            )
            if truck and truck.get("status") == "active":
                violations.append(
                    f"Truck {inspection['truck_id']}: defects reported on "
                    f"{inspection['inspection_date']} but truck not removed from service"
                )
    assert not violations, (
        f"VIOLATION: {len(violations)} defective PIT(s) not removed from service:\n"
        + "\n".join(violations)
    )

## Notes for the registry

- **Truck type specificity:** Training must cover the *specific* type of truck the operator will use (sit-down counterbalanced, stand-up rider, reach truck, order picker, etc.). A single "forklift certification" does not cover all types. The `truck_type_covered` field in training records is critical.
- **Refueling/recharging:** (l)(2)(i) prohibits fuel and battery operations in areas other than designated locations. A test against `fuel_change_location.is_designated == true` is recommended.
- **LP gas/propane trucks:** (c) requires that LP gas trucks be a specific type (Type GS or LP). LP cylinder change areas must be designated and properly ventilated. This is a DETERMINISTIC test against the truck approval and facility designation records.
- **Stability:** The operator must never exceed rated capacity. A test verifying that load records (if maintained) do not exceed rated capacity adds value in warehousing environments.
