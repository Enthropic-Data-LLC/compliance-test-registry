## Scope summary

1910.1200 requires chemical manufacturers, importers, and downstream employers to communicate chemical hazards through a written program, Safety Data Sheets (SDSs), container labels, and employee training. It incorporates the Globally Harmonized System (GHS) for classification and labeling. Almost every manufacturing, laboratory, cleaning, and maintenance operation is in scope. The 2024 revision aligns with GHS Rev. 7 — verify enforcement date for updated pictogram and SDS format requirements.

```python
"""
1910.1200(e) — Written Hazard Communication Program
Confidence: HIGH | Human Review: NOT REQUIRED

from datetime import datetime, timedelta

PROGRAM_REVIEW_INTERVAL_DAYS = 365


def test_hazcom_written_program_exists(written_programs):
    """Written HazCom program must exist and be on file."""
    prog = next(
        (p for p in written_programs if p["standard_id"] == "1910.1200"), None
    )
    assert prog is not None, (
        "VIOLATION: No written Hazard Communication Program found for 1910.1200."
    )


def test_hazcom_written_program_current(written_programs):
    """Written HazCom program reviewed within the last 12 months."""
    prog = next(p for p in written_programs if p["standard_id"] == "1910.1200")
    age = datetime.utcnow().date() - prog["last_reviewed_date"]
    assert age.days <= PROGRAM_REVIEW_INTERVAL_DAYS, (
        f"VIOLATION: HazCom written program last reviewed {age.days} days ago. "
        f"Threshold: {PROGRAM_REVIEW_INTERVAL_DAYS} days. "
        f"Review and update required."
    )


def test_hazcom_program_covers_required_elements(written_programs):
    """Written program must address labels, SDSs, and training."""
    prog = next(p for p in written_programs if p["standard_id"] == "1910.1200")
    required_sections = {"labels", "sds", "training"}
    covered = set(prog.get("sections_covered", []))
    missing = required_sections - covered
    assert not missing, (
        f"VIOLATION: HazCom program is missing required sections: {missing}. "
        f"Program must address labels (f), SDSs (g), and training (h)."
    )
```

```python
"""
1910.1200(g) — Safety Data Sheets
Confidence: HIGH | Human Review: NOT REQUIRED
"""

SDS_REQUIRED_SECTIONS = 16


def test_sds_exists_for_all_inventory_chemicals(chemical_inventory, sds_records):
    """Every hazardous chemical in inventory must have a corresponding SDS."""
    sds_ids = {s["chemical_id"] for s in sds_records}
    missing = [c for c in chemical_inventory if c["chemical_id"] not in sds_ids]
    assert not missing, (
        f"VIOLATION: {len(missing)} chemical(s) in inventory have no SDS: "
        f"{[c['chemical_id'] for c in missing]}"
    )


def test_sds_has_16_sections(sds_records):
    """Each SDS must contain all 16 GHS-required sections."""
    violations = [
        s for s in sds_records if s.get("sections_count", 0) < SDS_REQUIRED_SECTIONS
    ]
    assert not violations, (
        f"VIOLATION: {len(violations)} SDS record(s) have fewer than "
        f"{SDS_REQUIRED_SECTIONS} sections: "
        f"{[s['chemical_id'] for s in violations]}"
    )


def test_sds_in_english(sds_records):
    """SDS must be in English (additional languages permitted)."""
    non_english = [s for s in sds_records if s.get("language") != "en"]
    assert not non_english, (
        f"VIOLATION: {len(non_english)} SDS record(s) not in English: "
        f"{[s['chemical_id'] for s in non_english]}"
    )


def test_sds_accessible_all_shifts(sds_records):
    """SDSs must be accessible to employees on all shifts (paper or electronic)."""
    inaccessible = [s for s in sds_records if not s.get("accessible")]
    assert not inaccessible, (
        f"VIOLATION: {len(inaccessible)} SDS record(s) not marked as accessible: "
        f"{[s['chemical_id'] for s in inaccessible]}"
    )


def test_sds_not_superseded(sds_records, chemical_inventory):
    """SDS revision date must not predate the current product formulation date."""
    superseded = [
        s for s in sds_records
        if s.get("revision_date") and s.get("formulation_date")
        and s["revision_date"] < s["formulation_date"]
    ]
    assert not superseded, (
        f"VIOLATION: {len(superseded)} SDS record(s) predate their product "
        f"formulation: {[s['chemical_id'] for s in superseded]}"
    )
```

```python
"""
1910.1200(h) — Employee Information and Training
Confidence: HIGH | Human Review: NOT REQUIRED
"""


def test_hazcom_training_completed_before_exposure(employees, training_records):
    """All employees working with hazardous chemicals trained before or at assignment."""
    violations = []
    for emp in employees:
        if not emp.get("works_with_hazardous_chemicals"):
            continue
        records = [
            r for r in training_records
            if r["employee_id"] == emp["employee_id"]
            and r["standard_id"] == "1910.1200"
        ]
        if not records:
            violations.append(
                f"{emp['employee_id']} ({emp['job_title']}): no HazCom training on record"
            )
            continue
        earliest = min(r["training_date"] for r in records)
        if earliest > emp["work_area_assignment_date"]:
            violations.append(
                f"{emp['employee_id']}: trained {earliest} but "
                f"assigned {emp['work_area_assignment_date']}"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} employee(s) lack timely HazCom training:\n"
        + "\n".join(violations)
    )


def test_hazcom_training_covers_required_topics(training_records):
    """HazCom training records must document coverage of all required topics."""
    required_topics = {
        "employee_rights",
        "program_location",
        "sds_location",
        "detection_methods",
        "physical_health_hazards",
        "protective_measures",
    }
    violations = []
    for r in training_records:
        if r["standard_id"] != "1910.1200":
            continue
        covered = set(r.get("topics_covered", []))
        missing = required_topics - covered
        if missing:
            violations.append(
                f"{r['employee_id']} training on {r['training_date']}: "
                f"missing topics {missing}"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} HazCom training record(s) missing required topics:\n"
        + "\n".join(violations)
    )
```

## Notes for the registry

- **Watch item:** OSHA published an updated HazCom rule in 2024 aligning with GHS Rev. 7. Key changes affect SDS Section 2 format and pictogram assignments for certain hazard categories. Verify enforcement date; update label and SDS tests when effective.
- **No open assumptions.** The 16-section SDS structure and label elements are explicitly enumerated in the rule; no interpretation is required.
- **Highest citation frequency:** HazCom is OSHA's most-cited standard. The most common violations are: missing/incomplete SDSs for in-house formulations, labels removed/defaced on secondary containers, and training records that lack topic-level documentation.
- **Secondary container labeling:** (f)(6) requires immediate-use containers to be labeled unless the employee uses the chemical immediately and within their control. An additional test for secondary container labeling is recommended.
