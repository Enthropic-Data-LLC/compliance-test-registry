## Scope summary

1910.134 applies whenever respirators are necessary to protect the health of the employee or whenever the employer requires their use. It covers selection, training, medical evaluation, fit testing, use, maintenance, and program evaluation. Requirements differ for air-purifying respirators (APRs), atmosphere-supplying respirators (SCBAs, SARs), and tight-fitting vs. loose-fitting facepieces.

Two use categories have distinct requirements: **mandatory use** (required to protect health) and **voluntary use** (employer permits but does not require). This decomposition focuses on mandatory use; Appendix D covers the reduced obligations for voluntary use.

"""

Confidence: MEDIUM | Human Review: ASSUMPTION REQUIRED (ASSUME-134-002)
"""
import pytest


@pytest.mark.assumption(
    id="ASSUME-134-002",
    description="PLHCP clearance must reference the specific respirator model assigned",
    approved_by="Compliance Officer",
    review_date="2026-05-20",
)
def test_medical_clearance_before_respirator_use(
    respirator_assignments, medical_clearances
):
    violations = []
    for assignment in respirator_assignments:
        if not assignment.get("tight_fitting"):
            continue
        clearances = [
            c for c in medical_clearances
            if c["employee_id"] == assignment["employee_id"]
            and c.get("cleared")
        ]
        if not clearances:
            violations.append(
                f"{assignment['employee_id']}: assigned tight-fitting respirator "
                f"but no medical clearance on record"
            )
            continue
        most_recent = max(clearances, key=lambda c: c["clearance_date"])
        if most_recent["clearance_date"] > assignment["first_use_date"]:
            violations.append(
                f"{assignment['employee_id']}: medical clearance dated "
                f"{most_recent['clearance_date']} is AFTER first use "
                f"{assignment['first_use_date']}"
            )
        if most_recent.get("respirator_model_id") != assignment.get("facepiece_model_id"):
            violations.append(
                f"{assignment['employee_id']}: clearance references model "
                f"{most_recent.get('respirator_model_id')} but assigned "
                f"{assignment.get('facepiece_model_id')}"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} respirator assignment(s) lack valid medical "
        f"clearance:\n" + "\n".join(violations)

"""
1910.134(f) — Fit Testing (Annual)
Confidence: HIGH | Human Review: NOT REQUIRED
"""
from datetime import datetime, timedelta

FIT_TEST_INTERVAL_DAYS = 365


def test_fit_test_current_for_all_respirator_users(
    respirator_assignments, fit_test_records
):
    violations = []
    for assignment in respirator_assignments:
        if not assignment.get("tight_fitting"):
            continue
        relevant = [
            r for r in fit_test_records
            if r["employee_id"] == assignment["employee_id"]
            and r["facepiece_model_id"] == assignment["facepiece_model_id"]
            and r.get("pass")
        ]
        if not relevant:
            violations.append(
                f"{assignment['employee_id']}: no passing fit test for "
                f"assigned facepiece {assignment['facepiece_model_id']}"
            )
            continue
        most_recent = max(relevant, key=lambda r: r["test_date"])
        age = datetime.utcnow().date() - most_recent["test_date"]
        if age.days > FIT_TEST_INTERVAL_DAYS:
            violations.append(
                f"{assignment['employee_id']}: fit test {age.days} days old "
                f"(threshold {FIT_TEST_INTERVAL_DAYS})"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} employee(s) with expired or missing fit test:\n"
        + "\n".join(violations)
    )


def test_fit_test_used_osha_approved_protocol(fit_test_records):
    approved_protocols = {
        "QLFT-Isoamyl Acetate",
        "QLFT-Saccharin Solution Aerosol",
        "QLFT-Bitrex",
        "QLFT-Irritant Smoke",
        "QNFT-Generated Aerosol",
        "QNFT-Condensation Nuclei Counter",
        "QNFT-Controlled Negative Pressure",
    }
    violations = [
        r for r in fit_test_records
        if r.get("protocol") not in approved_protocols
        and not r.get("osha_equivalency_approved")
    ]
    assert not violations, (
        f"VIOLATION: {len(violations)} fit test record(s) used a non-approved protocol "
        f"without OSHA equivalency documentation: "
        f"{[r.get('employee_id') for r in violations]}"
    )


def test_fit_test_on_assigned_model(respirator_assignments, fit_test_records):
    violations = []
    for assignment in respirator_assignments:
        if not assignment.get("tight_fitting"):
            continue
        mismatched = [
            r for r in fit_test_records
            if r["employee_id"] == assignment["employee_id"]
            and r.get("pass")
            and r["facepiece_model_id"] != assignment["facepiece_model_id"]
        ]
        if mismatched and not any(
            r["facepiece_model_id"] == assignment["facepiece_model_id"]
            for r in fit_test_records
            if r["employee_id"] == assignment["employee_id"] and r.get("pass")
        ):
            violations.append(
                f"{assignment['employee_id']}: fit tested on "
                f"{[r['facepiece_model_id'] for r in mismatched]} "
                f"but assigned {assignment['facepiece_model_id']}"
            )
    assert not violations, (
        f"VIOLATION: {len(violations)} employee(s) fit tested on a different "
        f"model than assigned:\n" + "\n".join(violations)
    )

## Notes for the registry

- **Voluntary use (Appendix D):** When an employer permits but does not require respirator use, the only obligation is to distribute Appendix D (informational guidance) to the employee. No fit testing, medical evaluation, or written program is required for filtering facepiece respirators used voluntarily. A separate minimal test for Appendix D distribution may be warranted.
- **SCBA monthly inspection:** Self-contained breathing apparatus used for emergency escape must be inspected monthly. Add a test against `equipment_inspection.equipment_type == "SCBA"` and `inspection_date within 31 days`.
- **Air quality for atmosphere-supplying respirators:** (i)(1) requires Grade D breathing air. Test against `air_quality_record.grade == "D"` and `cylinders_labeled_not_oxygen`.
- **Recordkeeping (m):** Medical evaluations and fit test records are subject to HIPAA-equivalent confidentiality. Records must be retained per 1910.1020 (employee medical records: 30 years after employment ends).
